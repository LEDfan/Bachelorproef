#============================================================================
# Install the python test scripts.
#============================================================================
set (SCRIPTS
	testSimple.py
)

INSTALL (FILES
	${SCRIPTS}
	DESTINATION ${BIN_INSTALL_LOCATION}
)

#============================================================================
# Define tests.
#============================================================================
add_test(NAME simple_py
	WORKING_DIRECTORY ${TESTS_DIR}
	COMMAND python3 ${CMAKE_INSTALL_PREFIX}/${BIN_INSTALL_LOCATION}/testSimple.py
	)

#============================================================================
# Clean up.
#============================================================================
unset(SCRIPTS)
