124  /// Initialize beliefs container (including this in SetBeliefPolicy function slows you down
125  /// due to guarding aginst data races in parallel use of SetBeliefPolicy. The DoubleChecked
126  /// locking did not work in OpenMP parallel for's on Mac OSX.
127  template <typename BeliefPolicy>
128  void InitBeliefPolicy()
129  {
130  if (!m_beliefs) {
131  m_beliefs.emplace<util::SegmentedVector<BeliefPolicy>>(this->size());
132  } else {
133  throw std::runtime_error("_func_ : Error, already initialized!");
134  }
135  }

137  /// Assign the belief policy.
138  /// \tparam BeliefPolicy Template type param (we could use plain overloading here, i guess)
139  /// \param belief belief object that wille be associated with the person
140  /// \param i subscript to person associated with this belief object
141  // Cannot follow my preference for declaration of required explicit specializations, because SWIG
142  // does not like that. Hence include of the template method definition in the header file.
143  template <typename BeliefPolicy>
