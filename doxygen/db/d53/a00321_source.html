35 void Immunizer::Random(const util::SegmentedVector<ContactPool>& pools, std::vector<double>& immunityDistribution,
36  double immunityLinkProbability)
37 {
38  // Initialize a vector to count the population per age class [0-100].
39  std::vector<double> populationBrackets(100, 0.0);
40 
41  // Count individuals per age class and set all "susceptible" individuals "immune".
42  // note: focusing on measles, we expect the number of susceptible individuals
43  // to be less compared to the number of immune.
44  // TODO but this is a generic simulator
45  for (auto& c : pools) {
46  for (const auto& p : c.GetPool()) {
47  if (p->GetHealth().IsSusceptible()) {
48  p->GetHealth().SetImmune();
49  populationBrackets[p->GetAge()]++;
50  }
51  }
52  }
53 
54  // Sampler for int in [0, pools.size()) and for double in [0.0, 1.0).
55  const auto poolsSize = static_cast<int>(pools.size());
56  auto intGenerator = m_rn_manager.GetGenerator(trng::uniform_int_dist(0, poolsSize));
57  auto uniform01Generator = m_rn_manager.GetGenerator(trng::uniform01_dist<double>());
58 
59  // Calculate the number of susceptible individuals per age class.
60  unsigned int numSusceptible = 0;
61  for (unsigned int age = 0; age < 100; age++) {
62  populationBrackets[age] = floor(populationBrackets[age] * (1 - immunityDistribution[age]));
63  numSusceptible += populationBrackets[age];
64  }
65 
66  // Sample susceptible individuals, until all age-dependent quota are reached.
67  while (numSusceptible > 0) {
68  // random pool, random order of members
69  const ContactPool& p_pool = pools[intGenerator()];
70  const auto size = static_cast<unsigned int>(p_pool.GetSize());
71  std::vector<unsigned int> indices(size);
72  for (unsigned int i = 0; i < size; i++) {
73  indices[i] = i;
74  }
75  m_rn_manager.RandomShuffle(indices.begin(), indices.end());
76 
77  // loop over members, in random order
78  for (unsigned int i_p = 0; i_p < size && numSusceptible > 0; i_p++) {
79  Person& p = *p_pool.GetMember(indices[i_p]);
80  // if p is immune and his/her age class has not reached the quota => make susceptible
81  if (p.GetHealth().IsImmune() && populationBrackets[p.GetAge()] > 0) {
82  p.GetHealth().SetSusceptible();
83  populationBrackets[p.GetAge()]--;
84  numSusceptible--;
85  }
86  // random draw to continue in this pool or to sample a new one
87  if (uniform01Generator() < (1 - immunityLinkProbability)) {
88  break;
89  }
90  }
91  }
92 }
