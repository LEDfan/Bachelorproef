38 std::shared_ptr<Population> PopulationBuilder::Build(const ptree& config_pt, util::RNManager& /*rn_manager*/)
39 {
40  // ------------------------------------------------
41  // Setup.
42  // ------------------------------------------------
43  const auto pop = make_shared<Population>();
44  Population& population = *pop;
45  const auto belief_pt = config_pt.get_child("run.belief_policy");
46 
47  //------------------------------------------------
48  // Seeding rate.
49  //------------------------------------------------
50  const auto seeding_rate = config_pt.get<double>("run.seeding_rate");
51  if (seeding_rate > 1.0) {
52  throw runtime_error(string(__func__) + "> Bad input data.");
53  }
54 
55  //------------------------------------------------
56  // Build population from file.
57  //------------------------------------------------
58  const auto file_name = config_pt.get<string>("run.population_file");
59  const auto use_install_dirs = config_pt.get<bool>("run.use_install_dirs");
60  const auto file_path = (use_install_dirs) ? FileSys::GetDataDir() /= file_name : file_name;
61  if (!is_regular_file(file_path)) {
62  throw runtime_error(string(__func__) + "> Population file " + file_path.string() + " not present.");
63  }
64 
65  boost::filesystem::ifstream pop_file;
66  pop_file.open(file_path.string());
67  if (!pop_file.is_open()) {
68  throw runtime_error(string(__func__) + "> Error opening population file " + file_path.string());
69  }
70 
71  string line;
72  getline(pop_file, line); // step over file header
73  unsigned int person_id = 0U;
74 
75  while (getline(pop_file, line)) {
76  const auto values = Split(line, ",");
77  const auto risk_averseness = (values.size() <= 6) ? 0.0 : FromString<double>(values[6]);
78  const auto age = FromString<unsigned int>(values[0]);
79  const auto household_id = FromString<unsigned int>(values[1]);
80  const auto school_id = FromString<unsigned int>(values[2]);
81  const auto work_id = FromString<unsigned int>(values[3]);
82  const auto primary_community_id = FromString<unsigned int>(values[4]);
83  const auto secondary_community_id = FromString<unsigned int>(values[5]);
84 
85  population.CreatePerson(person_id, age, household_id, school_id, work_id, primary_community_id,
86  secondary_community_id, Health(), belief_pt, risk_averseness);
87  ++person_id;
88  }
89 
90  pop_file.close();
91 
92  //------------------------------------------------
93  // Done
94  //------------------------------------------------
95  return pop;
96 }
