60 std::shared_ptr<Population> Population::Create(const boost::property_tree::ptree& configPt, util::RNManager& rnManager)
61 {
62  // --------------------------------------------------------------
63  // Create (empty) population & and give it a ContactLogger.
64  // --------------------------------------------------------------
65  struct make_shared_enabler : public Population
66  {
67  };
68  auto pop = make_shared<make_shared_enabler>();
69  if (configPt.get<bool>("run.contact_output_file", true)) {
70  const auto prefix = configPt.get<string>("run.output_prefix");
71  const auto logPath = FileSys::BuildPath(prefix, "contact_log.txt");
72  pop->GetContactLogger() = LogUtils::CreateRotatingLogger("contact_logger", logPath.string());
73  pop->GetContactLogger()->set_pattern("%v");
74  } else {
75  pop->GetContactLogger() = LogUtils::CreateNullLogger("contact_logger");
76  }
77 
78  if (spdlog::get("travel_logger") == nullptr) {
79  if (configPt.get<bool>("run.travel_output_file", true)) {
80  const auto prefix = configPt.get<string>("run.output_prefix");
81  spdlog::register_logger(LogUtils::CreateFileLogger(
82  "travel_logger", FileSys::BuildPath(prefix, "travel_log.txt").string()));
83  } else {
84  spdlog::register_logger(LogUtils::CreateNullLogger("travel_logger"));
85  }
86  }
87 
88  pop->m_belief_pt = configPt.get_child("run.belief_policy");
89 
90  auto stride_logger = spdlog::get("stride_logger");
91 
92  boost::optional<const ptree&> regions = configPt.get_child_optional("run.regions");
93 
94  // -----------------------------------------------------------------------------------------
95  // Build population for each region
96  // -----------------------------------------------------------------------------------------
97  if (!regions) {
98  pop->m_regions["Default"] = 0;
99  pop->m_regionTravellerIndex.emplace_back();
100 
101  std::string geopop_type = configPt.get<std::string>("run.geopop_type", "default");
102 
103  CreateRegion(geopop_type, configPt, configPt.get_child("run"), pop, "Default", rnManager);
104  } else {
105  std::size_t currentId = 0;
106  for (const auto& region : configPt.get_child("run.regions")) {
107  std::string name = region.second.get<std::string>("name");
108  pop->m_regions[name] = currentId;
109  pop->m_regionTravellerIndex.emplace_back();
110  currentId++;
111  }
112  for (const auto& region : configPt.get_child("run.regions")) {
113  std::string name = region.second.get<std::string>("name");
114  std::string geopop_type = region.second.get<std::string>("geopop_type", "default");
115 
116  CreateRegion(geopop_type, configPt, region.second, pop, name, rnManager);
117  }
118  }
119 
120  return pop;
121 }

175 void Population::UpdateRegion(std::size_t regionId)
176 {
177  auto update = [this](std::size_t regionId) {
178  m_regionRanges.SetRange(size(), regionId);
179  for (auto id : ContactPoolType::IdList) {
180  m_pool_sys_regions[id].SetRange(m_pool_sys[id].size(), regionId);
181  }
182  m_currentRegionId = regionId;
183  m_have_inserted = true;
184  };
185 
186  if (m_currentRegionId != regionId || (!m_have_inserted && m_currentRegionId == 0)) {
187  if (!m_have_inserted) {
188  update(0);
189  }
190  for (std::size_t regId = m_currentRegionId + 1; regId <= regionId; regId++) {
191  update(regId);
192  }
193  }
194 }

220 void Population::CreateRegion(const std::string& geopop_type, const boost::property_tree::ptree& configPt,
221  const boost::property_tree::ptree& regionPt, const std::shared_ptr<Population>& pop,
222  const std::string& name, stride::util::RNManager& rnManager)
223 {
224  auto stride_logger = spdlog::get("stride_logger");
225 
226  if (geopop_type == "import") {
227  if (stride_logger)
228  stride_logger->info("Creating region \"{}\" with imported pop.", name);
229  ImportPopBuilder(configPt, regionPt, rnManager).Build(pop, pop->m_regions[name], name);
230  } else if (geopop_type == "generate") {
231  if (stride_logger)
232  stride_logger->info("Creating region \"{}\" with generated pop.", name);
233  GenPopBuilder(configPt, regionPt, rnManager).Build(pop, pop->m_regions[name], name);
234  } else {
235  if (stride_logger)
236  stride_logger->info("Creating region \"{}\" with Default pop.", name);
237  DefaultPopBuilder(configPt, regionPt, rnManager).Build(pop, pop->m_regions[name], name);
238  }
