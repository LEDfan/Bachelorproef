45 std::tuple<bool, size_t> ContactPool::SortMembers()
46 {
47  bool infectious_cases = false;
48  size_t num_cases = 0;
49 
50  for (size_t i_member = 0; i_member < m_index_immune; i_member++) {
51  // if immune, move to back
52  if (m_members[i_member]->GetHealth().IsImmune()) {
53  bool swapped = false;
54  size_t new_place = m_index_immune - 1;
55  m_index_immune--;
56  while (!swapped && new_place > i_member) {
57  if (m_members[new_place]->GetHealth().IsImmune()) {
58  m_index_immune--;
59  new_place--;
60  } else {
61  Swap(i_member, new_place);
62  swapped = true;
63  }
64  }
65  }
66  // else, if not susceptible, move to front
67  else if (!m_members[i_member]->GetHealth().IsSusceptible()) {
68  if (!infectious_cases && m_members[i_member]->GetHealth().IsInfectious()) {
69  infectious_cases = true;
70  }
71  if (i_member > num_cases) {
72  Swap(i_member, num_cases);
73  }
74  num_cases++;
75  }
76  }
77 
78  return std::make_tuple(infectious_cases, num_cases);
79 }

99 void ContactPool::RemoveMember(Person* person)
100 {
101  // use std::find and not std::remove since we need the original position to update the expats
102  // note that a person shouldn't be added to a CP more than once
103  auto it = std::find(m_members.begin(), m_members.end(), person);
104  if (it != m_members.end()) {
105  m_members.erase(it);
106  UpdateExpatsAfterRemoval(it);
107  }
108 }

110 void ContactPool::UpdateExpatsAfterRemoval(std::vector<Person*>::iterator itAfterRemovedPerson)
111 {
112  long index = itAfterRemovedPerson - m_members.begin();
113  for (std::pair<const Person* const, std::size_t>& expat : m_expats) {
114  if (expat.second >= static_cast<std::size_t>(index)) {
115  expat.second--;
116  }
117  assert(m_members[m_expats[expat.first]] == expat.first);
118  }

139 void ContactPool::Swap(std::size_t person1, std::size_t person2)
140 {
141  std::swap(m_members[person1], m_members[person2]);
142  Person* pPerson1 = m_members[person1];
143  if (pPerson1->IsTravelling() && pPerson1->GetPoolId(m_pool_type) == m_pool_id) {
144  // m_members[i_member] is an expat -> update map
145  m_expats[pPerson1] = person1;
146  assert(m_members[m_expats[pPerson1]] == pPerson1);
147  }
148  Person* pPerson2 = m_members[person2];
149  if (pPerson2->IsTravelling() && pPerson2->GetPoolId(m_pool_type) == m_pool_id) {
150  // m_members[i_member] is an expat -> update map
151  m_expats[pPerson2] = person2;
152  assert(m_members[m_expats[pPerson2]] == pPerson2);
153  }
154 }
