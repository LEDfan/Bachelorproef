123 void CSV::ReadFromStream(std::istream& inputStream)
124 {
125  std::string line;
126 
127  // header
128  getline(inputStream, line);
129  line = Trim(line);
130  std::vector<std::string> headerLabels = Split(line, ","); // Split is bad! There is no option to escape ",".
131  for (const std::string& label : headerLabels) {
132  labels.push_back(Trim(label, "\""));
133  }
134  columnCount = labels.size();
135 
136  // body
137  while (getline(inputStream, line)) {
138  line = Trim(line);
139  if (!line.empty()) {
140  std::vector<std::string> values =
141  Split(line, ","); // Split is bad! There is no option to escape ",".
142  AddRow(values);
143  }
144  }
145 }
