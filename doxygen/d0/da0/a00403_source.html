42  explicit SliceIndexer(T& t) : m_map(), m_ranges(), m_t(t) {}

44  /// Set a range. Warning: range is [ibegin, iend) i.e. half-open, iend not included!
45  range_type& SetSlice(std::size_t ibegin, std::size_t iend, const std::string& name)
46  {
47  if (m_map.find(name) != m_map.end()) {
48  throw std::range_error("SliceIndexer::SetSlice> Name is a duplicate: " + name);
49  } else {
50  m_ranges.emplace_back(range_type(m_t, ibegin, iend));
51  m_map[name] = m_ranges.size() - 1;
52  }
53  return m_ranges.back();
54  }
55 
56  /// Set a range, where the end is the end of the container.
57  range_type& SetSlice(std::size_t ibegin, const std::string& name)
58  {
59  if (m_map.find(name) != m_map.end()) {
60  throw std::range_error("SliceIndexer::SetSlice> Name is a duplicate: " + name);
61  } else {
62  m_ranges.emplace_back(range_type(m_t, ibegin, m_t.size()));
63  m_map[name] = m_ranges.size() - 1;
64  }
65  return m_ranges.back();
66  }
