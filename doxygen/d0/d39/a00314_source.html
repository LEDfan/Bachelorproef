34 /**
35  * An std::array modified to enable subscripting with the constact pool
36  * type indentifiers. The bounds checking "at" method, to protevt against
37  * a subscript that is static_cast<Id> from an erroneous integer value.
38  *
39  * We intentionally shadow the direct subscripting of the base array.
40  *
41  * Writing constructor is somehat weird since std::array has no constructors,
42  * only aggregate initialization. But we want some constructors for
43  * shorthand expressions.
44  *
45  * @tparam T base type of the array.
46  */
47 template <class T>
48 class IdSubscriptArray : public std::array<T, NumOfTypes()>
49 {
50 public:
51  /// What we 'll use most often and where we can have a default and
52  /// initialize all array elements to the same value.
53  /// e.g. IdSubscriptArray<unsigned int> m(1U);
54  explicit IdSubscriptArray(T t)
55  {
56  for (auto typ : IdList) {
57  this->operator[](typ) = t;
58  }
59  }
60 
61  explicit IdSubscriptArray()
62  {
63  for (auto typ : IdList) {
64  this->operator[](typ) = T();
65  }
66  }
67 
68  /// When we want to use an initializer list the elements is the
69  /// (possibly emty) initializer list are applied to the first
70  /// elements in order; any remaininig elements are default initialized.
71  /// e.g. IdSubscriptArray<unsigned int> mm {1U, 2U, 2U};
72  IdSubscriptArray(std::initializer_list<T> l)
73  {
74  auto it = l.begin();
75  for (auto typ : IdList) {
76  if (it != l.end()) {
77  this->operator[](typ) = *it;
78  ++it;
79  } else {
80  this->operator[](typ) = T();
81  }
82  }
83  }
84 
85  /// This actually works in itself but interferes annoyingly with the first
86  /// constructor above and is for practical purpose redundant.
87  ///
88  /// Forward constructor arguments to base class.
89  /// template <typename... Args>
90  /// explicit IdSubscriptArray(Args&&... args) : std::array<T, NumOfTypes()>{{std::forward<Args>(args)...}}{}
91 
92  /// Subscripting with pool typ id as argument.
93  typename std::array<T, NumOfTypes()>::reference operator[](ContactPoolType::Id id)
94  {
95  return this->std::template array<T, NumOfTypes()>::operator[](ContactPoolType::ToSizeT(id));
96  }
97 
98  /// Subscripting with pool typ id as argument.
99  typename std::array<T, NumOfTypes()>::const_reference operator[](ContactPoolType::Id id) const
100  {
101  return this->std::template array<T, NumOfTypes()>::operator[](ContactPoolType::ToSizeT(id));
102  }
103 
104  /// Subscripting with pool typ id as argument.
105  typename std::array<T, NumOfTypes()>::reference at(ContactPoolType::Id id)
106  {
107  if (ToSizeT(id) >= NumOfTypes()) {
108  throw std::out_of_range("IdSubscriptArray::at> Id out of range");
109  }
110  return this->std::template array<T, NumOfTypes()>::operator[](ContactPoolType::ToSizeT(id));
111  }
112 
113  /// Subscripting with pool typ id as argument.
114  typename std::array<T, NumOfTypes()>::const_reference at(ContactPoolType::Id id) const
115  {
116  if (ToSizeT(id) >= NumOfTypes()) {
117  throw std::out_of_range("IdSubscriptArray::at> Id out of range");
118  }
119  return this->std::template array<T, NumOfTypes()>::operator[](ContactPoolType::ToSizeT(id));
120  }
121 };
