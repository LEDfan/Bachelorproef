46 void SimRunner::Run(unsigned int numSteps)
47 {
48  // Saveguard against repeatedly firing AtStart, so bypass if numSteps == 0.
49  if (numSteps != 0U) {
50  // Prelims.
51  m_clock.Start();
52  const auto numDays = m_config_pt.get<unsigned int>("run.num_days");
53 
54  // We are AtStart: no steps have taken yet, so signal AtStart.
55  if (m_sim->GetCalendar()->GetSimulationDay() == 0) {
56  Notify(Id::AtStart);
57  }
58 
59  // Tahe numSteps but do not go beyond numDays.
60  for (unsigned int i = 0; i < numSteps; i++) {
61  // This is not the last step: execute and signal Stepped.
62  if (m_sim->GetCalendar()->GetSimulationDay() < numDays - 1) {
63  m_sim->TimeStep();
64  Notify(Id::Stepped);
65  // This is the last step so execute and afterwards signal Stepped and Finished
66  } else if (m_sim->GetCalendar()->GetSimulationDay() == numDays - 1) {
67  m_sim->TimeStep();
68  Notify(Id::Stepped);
69  Notify(Id::Finished);
70  break;
71  // We are apparently already at the end of the numDays so nothing to do or signal.
72  } else {
73  break;
74  }
75  }
76 
77  m_clock.Stop();
78  }
79 }
