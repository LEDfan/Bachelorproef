39 std::shared_ptr<Population> Population::Create(const boost::property_tree::ptree& configPt, util::RNManager& rnManager)
40 {
41  // --------------------------------------------------------------
42  // Create (empty) population & and give it a ContactLogger.
43  // --------------------------------------------------------------
44  struct make_shared_enabler : public Population
45  {
46  };
47  auto pop = make_shared<make_shared_enabler>();
48  if (configPt.get<bool>("run.contact_output_file", true)) {
49  const auto prefix = configPt.get<string>("run.output_prefix");
50  const auto logPath = FileSys::BuildPath(prefix, "contact_log.txt");
51  pop->GetContactLogger() = LogUtils::CreateRotatingLogger("contact_logger", logPath.string());
52  pop->GetContactLogger()->set_pattern("%v");
53  } else {
54  pop->GetContactLogger() = LogUtils::CreateNullLogger("contact_logger");
55  }
56 
57  pop->m_belief_pt = configPt.get_child("run.belief_policy");
58 
59  auto stride_logger = spdlog::get("stride_logger");
60 
61  boost::optional<const ptree&> regions = configPt.get_child_optional("run.regions");
62 
63  // -----------------------------------------------------------------------------------------
64  // Build population for each region
65  // -----------------------------------------------------------------------------------------
66  if (!regions) {
67  pop->m_regions["Default"] = 0;
68 
69  std::string geopop_type = configPt.get<std::string>("run.geopop_type", "default");
70 
71  CreateRegion(geopop_type, configPt, configPt.get_child("run"), pop, "Default", rnManager);
72  } else {
73  std::size_t currentId = 0;
74  for (const auto& region : configPt.get_child("run.regions")) {
75  std::string name = region.second.get<std::string>("name");
76  pop->m_regions[name] = currentId;
77  currentId++;
78  }
79  for (const auto& region : configPt.get_child("run.regions")) {
80  std::string name = region.second.get<std::string>("name");
81  std::string geopop_type = region.second.get<std::string>("geopop_type", "default");
82 
83  CreateRegion(geopop_type, configPt, region.second, pop, name, rnManager);
84  }
85  }
86 
87  return pop;
88 }

165 void Population::CreateRegion(const std::string& geopop_type, const boost::property_tree::ptree& configPt,
166  const boost::property_tree::ptree& regionPt, const std::shared_ptr<Population>& pop,
167  const std::string& name, stride::util::RNManager& rnManager)
168 {
169  auto stride_logger = spdlog::get("stride_logger");
170 
171  if (geopop_type == "import") {
172  if (stride_logger)
173  stride_logger->info("Creating region \"{}\" with imported pop.", name);
174  ImportPopBuilder(configPt, regionPt, rnManager).Build(pop, pop->m_regions[name], name);
175  } else if (geopop_type == "generate") {
176  if (stride_logger)
177  stride_logger->info("Creating region \"{}\" with generated pop.", name);
178  GenPopBuilder(configPt, regionPt, rnManager).Build(pop, pop->m_regions[name], name);
179  } else {
180  if (stride_logger)
181  stride_logger->info("Creating region \"{}\" with Default pop.", name);
182  DefaultPopBuilder(configPt, regionPt, rnManager).Build(pop, pop->m_regions[name], name);
183  }
