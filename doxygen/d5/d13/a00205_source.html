93 std::pair<double, double> Calibrator::FindMeanStdev(std::vector<unsigned int> results) const
94 {
95  double sum = std::accumulate(results.begin(), results.end(), 0.0);
96  double mean = sum / results.size();
97 
98  // Inspired by https://stackoverflow.com/a/7616783
99  std::vector<double> diff(results.size());
100  std::transform(results.begin(), results.end(), diff.begin(), [mean](double x) { return x - mean; });
101  double sq_sum = std::inner_product(diff.begin(), diff.end(), diff.begin(), 0.0);
102  double stdev = std::sqrt(sq_sum / results.size());
103  return std::make_pair(mean, stdev);
104 }
