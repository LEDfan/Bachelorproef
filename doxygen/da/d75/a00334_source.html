127 // Definition for primary template covers the situation for LogMode::None & LogMode::Transmissions,
128 // both with track_index_case false and true.
129 // And every local information policy except NoLocalInformation
130 //-------------------------------------------------------------------------------------------------
131 template <LogMode::Id LL, bool TIC, typename LIP, bool TO>
132 void Infector<LL, TIC, LIP, TO>::Exec(ContactPool& pool, DiseaseProfile disease_profile, ContactHandler contact_handler,
133  shared_ptr<const Calendar> calendar)
134 {
135  using LP = LOG_POLICY<LL>;
136  using RP = R0_POLICY<TIC>;
137 
138  pool.UpdateMemberPresence();
139 
140  // set up some stuff
141  auto logger = spdlog::get("contact_logger");
142  const auto c_type = pool.m_pool_type;
143  const auto& c_members = pool.m_members;
144  const auto t_rate = disease_profile.GetTransmissionRate();
145 
146  // check all contacts
147  for (size_t i_person1 = 0; i_person1 < c_members.size(); i_person1++) {
148  // check if member is present today
149  if (c_members[i_person1].second) {
150  auto p1 = c_members[i_person1].first;
151  const double c_rate = pool.GetContactRate(p1);
152  // loop over possible contacts (contacts can be initiated by each member)
153  for (size_t i_person2 = 0; i_person2 < c_members.size(); i_person2++) {
154  // check if not the same person
155  if (i_person1 != i_person2) {
156  // check if member is present today
157  if (c_members[i_person2].second) {
158  auto p2 = c_members[i_person2].first;
159  // check for contact
160  if (contact_handler.HasContact(c_rate)) {
161  // log contact if person 1 is participating in survey
162  if (p1->IsParticipatingInSurvey()) {
163  LP::Contact(logger, p1, p2, c_type, calendar);
164  }
165  // log contact if person 2 is participating in survey
166  if (p2->IsParticipatingInSurvey()) {
167  LP::Contact(logger, p2, p1, c_type, calendar);
168  }
169 
170  // exchange info about health state & beliefs
171  LIP::Update(p1, p2);
172 
173  // transmission & infection.
174  if (contact_handler.HasTransmission(t_rate)) {
175  if (p1->GetHealth().IsInfectious() &&
176  p2->GetHealth().IsSusceptible()) {
177  LP::Transmission(logger, p1, p2, c_type,
178  calendar);
179  p2->GetHealth().StartInfection();
180  RP::Exec(p2);
181  } else if (p2->GetHealth().IsInfectious() &&
182  p1->GetHealth().IsSusceptible()) {
183  LP::Transmission(logger, p2, p1, c_type,
184  calendar);
185  p1->GetHealth().StartInfection();
186  RP::Exec(p1);
187  }
188  }
189  }
190  }
191  }
192  }
193  }
194  }

200 template <LogMode::Id LL, bool TIC>
201 void Infector<LL, TIC, NoLocalInformation, true>::Exec(ContactPool& pool, DiseaseProfile disease_profile,
202  ContactHandler ch, shared_ptr<const Calendar> calendar)
203 {
204  using LP = LOG_POLICY<LL>;
205  using RP = R0_POLICY<TIC>;
206 
207  // check for infected members and sort
208  bool infectious_cases;
209  size_t num_cases;
210  tie(infectious_cases, num_cases) = pool.SortMembers();
211 
212  if (infectious_cases) {
213  pool.UpdateMemberPresence();
214 
215  // set up some stuff
216  auto logger = spdlog::get("contact_logger");
217  const auto c_type = pool.m_pool_type;
218  const auto c_immune = pool.m_index_immune;
219  const auto& c_members = pool.m_members;
220  const auto t_rate = disease_profile.GetTransmissionRate();
221 
222  // match infectious and susceptible members, skip last part (immune members)
223  for (size_t i_infected = 0; i_infected < num_cases; i_infected++) {
224  // check if member is present today
225  if (c_members[i_infected].second) {
226  const auto p1 = c_members[i_infected].first;
227  if (p1->GetHealth().IsInfectious()) {
228  const double c_rate_p1 = pool.GetContactRate(p1);
229  // loop over possible susceptible contacts
230  for (size_t i_contact = num_cases; i_contact < c_immune; i_contact++) {
231  // check if member is present today
232  if (c_members[i_contact].second) {
233  auto p2 = c_members[i_contact].first;
234  const double c_rate_p2 = pool.GetContactRate(p2);
235  if (ch.HasContactAndTransmission(c_rate_p1, t_rate) ||
236  ch.HasContactAndTransmission(c_rate_p2, t_rate)) {
237  if (p1->GetHealth().IsInfectious() &&
238  p2->GetHealth().IsSusceptible()) {
239  p2->GetHealth().StartInfection();
240  RP::Exec(p2);
241  LP::Transmission(logger, p1, p2, c_type,
242  calendar);
243  }
244  }
245  }
246  }
247  }
248  }
249  }
250  }
