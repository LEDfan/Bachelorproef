46 void DiseaseSeeder::Seed(std::shared_ptr<Population> pop)
47 {
48  // --------------------------------------------------------------
49  // Population immunity (natural immunity & vaccination).
50  // --------------------------------------------------------------
51  const auto immunityProfile = m_config_pt.get<std::string>("run.immunity_profile");
52  Vaccinate("immunity", immunityProfile, pop->GetContactPoolSys()[Id::Household]);
53 
54  const auto vaccinationProfile = m_config_pt.get<std::string>("run.vaccine_profile");
55  Vaccinate("vaccine", vaccinationProfile, pop->GetContactPoolSys()[Id::Household]);
56 
57  // --------------------------------------------------------------
58  // Seed infected persons.
59  // --------------------------------------------------------------
60  const auto sRate = m_config_pt.get<double>("run.seeding_rate");
61  const auto sAgeMin = m_config_pt.get<double>("run.seeding_age_min", 1);
62  const auto sAgeMax = m_config_pt.get<double>("run.seeding_age_max", 99);
63  const auto popSize = pop->size();
64  const auto maxPopIndex = static_cast<unsigned int>(popSize - 1);
65  auto generator = m_rn_manager.GetGenerator(trng::uniform_int_dist(0, maxPopIndex));
66  auto& logger = pop->GetContactLogger();
67 
68  auto numInfected = static_cast<unsigned int>(floor(static_cast<double>(popSize) * sRate));
69  while (numInfected > 0) {
70  Person& p = pop->at(static_cast<size_t>(generator()));
71  if (p.GetHealth().IsSusceptible() && (p.GetAge() >= sAgeMin) && (p.GetAge() <= sAgeMax)) {
72  p.GetHealth().StartInfection();
73  numInfected--;
74  logger->info("[PRIM] {} {} {} {}", -1, p.GetId(), -1, 0);
75  }
76  }
77 }

79 void DiseaseSeeder::Vaccinate(const std::string& immunityType, const std::string& immunizationProfile,
80  util::SegmentedVector<ContactPool>& immunityPools)
81 {
82  std::vector<double> immunityDistribution;
83  const double linkProbability = 0;
84  Immunizer immunizer(m_rn_manager);
85 
86  if (immunizationProfile == "Random") {
87  const auto immunityRate = m_config_pt.get<double>("run." + ToLower(immunityType) + "_rate");
88  for (unsigned int index_age = 0; index_age < 100; index_age++) {
89  immunityDistribution.push_back(immunityRate);
90  }
91  immunizer.Random(immunityPools, immunityDistribution, linkProbability);
92  } else if (immunizationProfile == "Cocoon") {
93  immunizer.Cocoon(immunityPools, immunityDistribution, linkProbability);
94  }
95 }
