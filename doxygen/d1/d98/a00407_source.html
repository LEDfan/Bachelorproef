64 void Sim::TimeStep()
65 {
66  // Logic where you compute (on the basis of input/config for initial day or on the basis of
67  // number of sick persons, duration of epidemic etc) what kind of DaysOff scheme you apply.
68  const auto daysOff = std::make_shared<DaysOffStandard>(m_calendar);
69  const bool isWorkOff = daysOff->IsWorkOff();
70  const bool isSchoolOff = daysOff->IsSchoolOff();
71 
72  // To be used in update of population & contact pools.
73  Population& population = *m_population;
74  auto& poolSys = population.GetContactPoolSys();
75  auto contactLogger = population.GetContactLogger();
76  const auto simDay = m_calendar->GetSimulationDay();
77  const auto& infector = *m_infector;
78 
79 #pragma omp parallel num_threads(m_num_threads)
80  {
81  // Update health status and presence/absence in pools
82  // depending on health status, work/school day.
83 #pragma omp for schedule(static)
84  for (size_t i = 0; i < population.size(); ++i) {
85  population[i].Update(isWorkOff, isSchoolOff, m_travellerProfile, m_population);
86  }
87 
88  // Infector updates individuals for contacts & transmission within each pool.
89  // Skip pools with id = 0, because it means Not Applicable.
90  const auto thread_num = static_cast<unsigned int>(omp_get_thread_num());
91  for (auto typ : ContactPoolType::IdList) {
92  if ((typ == ContactPoolType::Id::Work && isWorkOff) ||
93  (typ == ContactPoolType::Id::K12School && isSchoolOff) ||
94  (typ == ContactPoolType::Id::College && isSchoolOff)) {
95  continue;
96  }
97 #pragma omp for schedule(static)
98  for (size_t i = 1; i < poolSys[typ].size(); i++) { // NOLINT
99  infector(poolSys[typ][i], m_contact_profiles[typ], m_transmission_profile,
100  m_handlers[thread_num], simDay, contactLogger);
101  }
102  }
103  }
104 
105  m_population->GetContactLogger()->flush();
106  m_calendar->AdvanceDay();
107 }
