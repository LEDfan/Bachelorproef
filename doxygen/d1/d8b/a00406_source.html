37 /**
38  * Container that stores objects "almost contiguously" (in a chain of blocks)
39  * and guarantees that pointers/iterators are not invalidated when the container
40  * grows. Elements are assigned to the container either sequentially through
41  * push_back and emplace_back or through direct adressing with emplace, at or
42  * the subscript operator.
43  * It supports most familiar operators except reserve (no need for advance
44  * reservation of capacity to avoid re-allocation that invalidates pointers as
45  * in std::vector) and insertion/deletion.
46  *
47  * Template parameters:
48  * T type of elements stored in the container
49  * N block size i.e. number of elements per block
50  */
51 template <typename T, size_t N = 512, bool Safe = true>
52 class SegmentedVector

215  /// Increases the number of elements (but DOES NOT INITIALIZE the additional
216  /// elements) or pops elements (and DOES RUN those element's destructor).
217  /// CAVEAT: if you resize (as you do here) but do not subsequently initialize all
218  /// elements, the SegmentedVector destructor or a call to clear will cause a segmentation
219  /// fault because of the destructor call on unitilialezed elements.
220  void resize(size_type new_size)
221  {
222  if (new_size < size()) {
223  if (Safe) {
224  for (size_type i = m_size - 1; new_size - 1 < i; --i) {
225  pop_back();
226  }
227  } else {
228  const size_type new_block_count = 1 + (new_size - 1) / N;
229  while (new_block_count < get_block_count()) {
230  delete[] m_blocks[m_blocks.size() - 1];
231  m_blocks.pop_back();
232  }
233  m_size = new_size;
234  }
235  } else if (new_size > size()) {
236  if (Safe) {
237  for (size_type i = size(); i < new_size; ++i) {
238  push_back(std::move(T()));
239  }
240  } else {
241  while (new_size > capacity()) {
242  m_blocks.push_back(new Chunk[N]);
243  }
244  m_size = new_size;
245  }
246  }
247  assert((size() == new_size));
248  assert((size() <= capacity()));
249  }
250 
251  void resize(size_type new_size, const value_type& value)
252  {
253  if (new_size < size()) {
254  resize(new_size);
255  } else if (new_size > size()) {
256  for (size_type i = size(); i < new_size; ++i) {
257  push_back(value);
258  }
259  }
260  assert((size() == new_size));
261  assert((size() <= capacity()));
262  }

264  /// Clears the content.
265  void clear()
266  {
267  if (Safe) {
268  for (auto& i : *this) {
269  i.~T();
270  }
271  }
272  for (auto p : m_blocks) {
273  delete[] p;
274  }
275  m_blocks.clear();
276  m_size = 0;
277  assert(size() == 0);
278  assert(m_blocks.size() == 0);
279  }

281  /// Constructs element in-place at position pos.
282  template <class... Args>
283  T* emplace(size_type pos, Args&&... args)
284  {
285  assert(0 <= pos && pos < m_size);
286  T* memory = static_cast<T*>(static_cast<void*>(&(m_blocks[pos / N][pos % N])));
287  return new (memory) T(std::forward<Args>(args)...); // construct new object
288  }
289 
290  /// Constructs element in-place at the end.
291  template <class... Args>
292  T* emplace_back(Args&&... args)
293  {
294  T* memory = this->get_chunk();
295  return new (memory) T(std::forward<Args>(args)...); // construct new object
296  }

298  /// Removes the last element.
299  void pop_back()
300  {
301  // No pop on empty container.
302  if (m_size <= 0) {
303  throw std::logic_error("CompactStorage::pop_back called on empty object.");
304  }
305 
306  // Element destruction.
307  at(m_size - 1).~T();
308  --m_size;
309 
310  // If tail block vacated, release it.
311  const size_t last_block_index = m_blocks.size() - 1;
312  if (m_size <= last_block_index * N) {
313  delete[] m_blocks[last_block_index];
314  m_blocks.pop_back();
315  }
316  }
317 
318  /// Adds element to end.
319  T* push_back(const T& obj)
320  {
321  T* memory = get_chunk();
322  return new (memory) T(obj); // copy-construct new object
323  }
324 
325  /// Adds element to end.
326  T* push_back(T&& obj)

328  T* memory = get_chunk();

340 private:
341  /// Get next available chunk for element construction with placement new.
342  T* get_chunk()
343  {
344  const size_t b = m_size / N; // Index of block in vector m_blocks
345  const size_t i = m_size % N; // Offset of chunk within its block
346 
347  if (b == m_blocks.size()) { // Out of buffers, last buffer is full
348  auto chunk = new Chunk[N];
349  m_blocks.push_back(chunk);
350  }
351  ++m_size;
352  return static_cast<T*>(static_cast<void*>(&((m_blocks[b])[i])));
353  }
354
