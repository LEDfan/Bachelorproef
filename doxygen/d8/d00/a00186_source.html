112 void BaseController::RegisterViewers()
113 {
114  // Command line viewer
115  m_stride_logger->info("Registering CliViewer");
116  const auto cli_v = make_shared<viewers::CliViewer>(m_runner, m_stride_logger);
117  m_runner->Register(cli_v, bind(&viewers::CliViewer::Update, cli_v, placeholders::_1));
118 
119  // Adopted viewer

124

127  m_stride_logger->info("Registering InfectedViewer");

130

144 void BaseController::LogSetup()
145 {
146  m_stride_logger->info("BaseController stating up at: {}", TimeStamp().ToString());
147  m_stride_logger->info("Executing revision {}", ConfigInfo::GitRevision());
148  m_stride_logger->info("Creating dir: {}", m_output_prefix);
149  m_stride_logger->trace("Executing: {}", FileSys::GetExecPath().string());
150  m_stride_logger->trace("Current directory: {}", FileSys::GetCurrentDir().string());
151  if (m_use_install_dirs) {
152  m_stride_logger->trace("Install directory: {}", FileSys::GetRootDir().string());
153  m_stride_logger->trace("Config directory: {}", FileSys::GetConfigDir().string());
154  m_stride_logger->trace("Data directory: {}", FileSys::GetDataDir().string());
155  }
156  if (ConfigInfo::HaveOpenMP()) {
157  m_stride_logger->info("Max number OpenMP threads in this environment: {}",
158  ConfigInfo::NumberAvailableThreads());
159  m_stride_logger->info("Configured number of threads: {}",
160  m_config_pt.get<unsigned int>("run.num_threads"));
161  } else {
162  m_stride_logger->info("Not using OpenMP threads.");
163  }
164 }
