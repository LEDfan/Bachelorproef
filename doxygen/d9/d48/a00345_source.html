42 /**
43  * A k-d tree: a k-dimensional generalization of binary search trees
44  * This data structure allows for efficient lookup of points and range queries with an Axis-Aligned Bounding Box (when
45  * balanced).
46  *
47  * The template parameter `P` should have the following attributes and operations:
48  * - A `static constexpr std::size_t dim`: the number of dimensions of the point type.
49  * - A `template <std::size_d D> get() const` method that returns the coordinate of the `D`th dimension of the point
50  * - A nested `template <std::size_t D> struct dimension_type` that has a member type `type` giving the return type for
51  * `Get<D>`
52  * - A default constructor and a copy constructor
53  * - The individual dimensions should each have a total order and equality
54  * - A method `bool InBox(const AABB<P>& box) const` that indicates if a point falls withing the bounding box (only for
55  * range queries)
56  */

358 std::size_t Median(const std::vector<P>& points)
359 {
360  if (points.empty())
361  return 0;
362

364  std::vector<C> sorting;
365  for (std::size_t i = 0; i < points.size(); i++) {
366  sorting.emplace_back(points[i].template Get<D>(), i);

369  return sorting[sorting.size() / 2].second;
