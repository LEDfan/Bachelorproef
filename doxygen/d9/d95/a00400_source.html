44 shared_ptr<Sim> SimBuilder::Build(shared_ptr<Sim> sim, shared_ptr<Population> pop)
45 {
46  // --------------------------------------------------------------
47  // Read config info and setup random number manager
48  // --------------------------------------------------------------
49  sim->m_config_pt = m_config_pt;
50  sim->m_population = std::move(pop);
51  sim->m_track_index_case = m_config_pt.get<bool>("run.track_index_case");
52  sim->m_num_threads = m_config_pt.get<unsigned int>("run.num_threads");
53  sim->m_calendar = make_shared<Calendar>(m_config_pt);
54  sim->m_local_info_policy = m_config_pt.get<string>("run.local_information_policy", "NoLocalInformation");
55  sim->m_contact_log_mode = ContactLogMode::ToMode(m_config_pt.get<string>("run.contact_log_level", "None"));
56 
57  // --------------------------------------------------------------
58  // Contact handlers, each with generator bound to different
59  // random engine stream) and infector.
60  // --------------------------------------------------------------
61  for (size_t i = 0; i < sim->m_num_threads; i++) {
62  auto gen = sim->m_rn_manager.GetGenerator(trng::uniform01_dist<double>(), i);
63  sim->m_handlers.emplace_back(ContactHandler(gen));
64  }
65  const auto& select = make_tuple(sim->m_contact_log_mode, sim->m_track_index_case, sim->m_local_info_policy);
66  sim->m_infector = InfectorMap().at(select);
67 
68  // --------------------------------------------------------------
69  // Initialize the age-related contact profiles.
70  // --------------------------------------------------------------
71  const auto ageContactPt = ReadAgeContactPtree();
72  for (Id typ : IdList) {
73  sim->m_contact_profiles[typ] = AgeContactProfile(typ, ageContactPt);
74  }
75 
76  // --------------------------------------------------------------
77  // Initialize the transmission profile (fixes rates).
78  // --------------------------------------------------------------
79  const auto diseasePt = ReadDiseasePtree();
80  sim->m_transmission_profile.Initialize(m_config_pt, diseasePt);
81 
82  // --------------------------------------------------------------
83  // Seed the population with health data.
84  // --------------------------------------------------------------
85  HealthSeeder(diseasePt).Seed(sim->m_population, sim->m_handlers);
86 
87  // --------------------------------------------------------------
88  // Seed population wrt immunity/vaccination/infection.
89  // --------------------------------------------------------------
90  DiseaseSeeder(m_config_pt, sim->m_rn_manager).Seed(sim->m_population);
91 
92  // --------------------------------------------------------------
93  // Seed population wrt belief policies.
94  // --------------------------------------------------------------
95  BeliefSeeder(m_config_pt, sim->m_rn_manager).Seed(sim->m_population);
96 
97  sim->m_travellerProfile = TravellerProfileBuilder(m_config_pt, sim->m_rn_manager, sim->m_population).Build();
98 
99  // --------------------------------------------------------------
100  // Done.
101  // --------------------------------------------------------------
102  return sim;
103 }
