76  /// Produce a random generator out of a distribution bound to sim's random engine.
77  /// @param d The distribution object.
78  /// @param i Which stream gets sampled, 0 by default.
79  /// @return The random generator g that can be called as g()
80  template <typename D>
81  std::function<typename D::result_type()> GetGenerator(const D& d, size_t i = 0)
82  {
83  using namespace RNEngineType;
84  std::function<typename D::result_type()> gen;
85  switch (m_type_id) {
86  case Id::lcg64: gen = std::bind(d, std::ref(m_lcg64[i])); break;
87  case Id::lcg64_shift: gen = std::bind(d, std::ref(m_lcg64_shift[i])); break;
88  case Id::mrg2: gen = std::bind(d, std::ref(m_mrg2[i])); break;
89  case Id::mrg3: gen = std::bind(d, std::ref(m_mrg3[i])); break;
90  case Id::yarn2: gen = std::bind(d, std::ref(m_yarn2[i])); break;
91  case Id::yarn3: gen = std::bind(d, std::ref(m_yarn3[i])); break;
92  }
93  return gen;
94  }

96  /// Randomly shuffle items between iterators using i-th stream of the random engine.
97  /// \param first
98  /// \param last
99  /// \return
100  template <typename RandomIt>
101  void RandomShuffle(RandomIt first, RandomIt last, size_t i = 0)
102  {
103  using namespace RNEngineType;
104  switch (m_type_id) {
105  case Id::lcg64: std::random_shuffle(first, last, m_lcg64[i]); break;
106  case Id::lcg64_shift: std::random_shuffle(first, last, m_lcg64_shift[i]); break;
107  case Id::mrg2: std::random_shuffle(first, last, m_mrg2[i]); break;
108  case Id::mrg3: std::random_shuffle(first, last, m_mrg3[i]); break;
109  case Id::yarn2: std::random_shuffle(first, last, m_yarn2[i]); break;
110  case Id::yarn3: std::random_shuffle(first, last, m_yarn3[i]); break;
111  }
112  }
