39 shared_ptr<Population> SurveySeeder::Seed(shared_ptr<Population> pop)
40 {
41  const string log_level = m_config_pt.get<string>("run.contact_log_level", "None");
42  if (log_level == "All" || log_level == "Susceptibles") {
43  Population& population = *pop;
44  auto& logger = population.GetContactLogger();
45  const auto max_index = static_cast<unsigned int>(population.size() - 1);
46  auto generator = m_rn_manager.GetGenerator(trng::uniform_int_dist(0, max_index));
47  const auto participants = m_config_pt.get<unsigned int>("run.num_participants_survey");
48 
49  // Use while-loop to get 'participants' unique participants (default sampling is with replacement).
50  // A for loop will not do because we might draw the same person twice.
51  auto num_samples = 0U;
52  while (num_samples < participants) {
53  Person& p = population[generator()];
54  if (!p.IsSurveyParticipant()) {
55  p.ParticipateInSurvey();
56  logger->info("[PART] {} {} {} {} {} {}", p.GetId(), p.GetAge(), p.GetGender(),
57  p.GetPoolId(Id::K12School), p.GetPoolId(Id::College),
58  p.GetPoolId(Id::Work));
59  num_samples++;
60  }
61  }
62  }
63  return pop;
64 }
